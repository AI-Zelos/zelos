zelos.coordination

Pluggable Coordination Backends — Distributed leader election and service discovery.

Phase 7: InMemory (default), etcd, and abstract base for future backends (Consul, ZooKeeper).

All backends implement the CoordinationBackend interface:
  • register_node / deregister_node
  • elect_leader (campaign)
  • get_leader / watch_leader
  • heartbeat (lease renewal)

 90class InMemoryCoordinationBackend(CoordinationBackend):
 91    """Phase 7 default: In-memory coordination for single-node deployments."""
 92
 93    def __init__(self, config: dict | None = None):
 94        super().__init__(config)
 95        self._nodes: dict[str, CoordinationNode] = {}
 96        self._leader_id: str | None = None
 97        self._leader_ttl: float = 0.0
 98        self._leader_elected_at: float = 0.0
 99        self._watchers: list[Callable] = []
100        self._lock = threading.RLock()
101        self._monitor_thread: threading.Thread | None = None
102
103    def connect(self) -> bool:
104        self._connected = True
105        return True
106
107    def disconnect(self) -> None:
108        self._connected = False
109        if self._monitor_thread:
110            self._monitor_thread = None
111
112    def health(self) -> bool:
113        return self._connected
114
115    def register_node(self, node: CoordinationNode) -> bool:
116        with self._lock:
117            self._nodes[node.node_id] = node
118        return True
119
120    def deregister_node(self, node_id: str) -> bool:
121        with self._lock:
122            if node_id in self._nodes:
123                del self._nodes[node_id]
124            if self._leader_id == node_id:
125                self._leader_id = None
126        return True
127
128    def elect_leader(self, node_id: str, ttl_seconds: float = 30.0) -> bool:
129        with self._lock:
130            if node_id not in self._nodes:
131                return False
132            # Lexicographically smallest node_id wins (Bully algorithm)
133            candidates = sorted(self._nodes.keys())
134            if candidates and candidates[0] == node_id:
135                self._leader_id = node_id
136                self._leader_ttl = ttl_seconds
137                self._leader_elected_at = time.time()
138                for w in self._watchers:
139                    try:
140                        w(node_id)
141                    except Exception:
142                        pass
143                return True
144        return False
145
146    def get_leader(self) -> str | None:
147        with self._lock:
148            if self._leader_id and time.time() - self._leader_elected_at > self._leader_ttl:
149                self._leader_id = None
150            return self._leader_id
151
152    def watch_leader(self, callback: Callable[[str | None], None]) -> None:
153        with self._lock:
154            self._watchers.append(callback)
155
156    def heartbeat(self, node_id: str) -> bool:
157        with self._lock:
158            if node_id == self._leader_id:
159                self._leader_elected_at = time.time()
160                node = self._nodes.get(node_id)
161                if node:
162                    node.last_heartbeat = time.time()
163                return True
164        return False
165
166    def list_nodes(self) -> list[CoordinationNode]:
167        with self._lock:
168            return list(self._nodes.values())

174class EtcdCoordinationBackend(CoordinationBackend):
175    """etcd v3 coordination backend.
176
177    Uses etcd's lease + transaction for leader election (campaign pattern).
178    Config:
179      endpoints: "localhost:2379" (or comma-separated list)
180      prefix: "/zelos/" (key prefix for all Zelos keys)
181    """
182
183    def __init__(self, config: dict | None = None):
184        super().__init__(config)
185        cfg = config or {}
186        self._endpoints = cfg.get("endpoints", "localhost:2379")
187        self._prefix = cfg.get("prefix", "/zelos/")
188        self._client = None
189        self._lease_id = None
190
191    def connect(self) -> bool:
192        try:
193            import etcd3
194            host, port = self._endpoints.split(":") if ":" in self._endpoints else (self._endpoints, "2379")
195            self._client = etcd3.client(host=host, port=int(port))
196            self._connected = True
197            return True
198        except ImportError:
199            return False  # etcd3 not installed
200        except Exception:
201            self._connected = False
202            return False
203
204    def disconnect(self) -> None:
205        if self._lease_id and self._client:
206            try:
207                self._client.revoke_lease(self._lease_id)
208            except Exception:
209                pass
210        self._connected = False
211        self._client = None
212
213    def health(self) -> bool:
214        if not self._client or not self._connected:
215            return False
216        try:
217            self._client.status()
218            return True
219        except Exception:
220            return False
221
222    def _key(self, name: str) -> str:
223        return f"{self._prefix}{name}"
224
225    def register_node(self, node: CoordinationNode) -> bool:
226        if not self._client:
227            return False
228        try:
229            data = json.dumps({
230                "node_id": node.node_id, "host": node.host, "port": node.port,
231                "capabilities": node.capabilities, "last_heartbeat": time.time(),
232            })
233            self._client.put(self._key(f"nodes/{node.node_id}"), data)
234            return True
235        except Exception:
236            return False
237
238    def deregister_node(self, node_id: str) -> bool:
239        if not self._client:
240            return False
241        try:
242            self._client.delete(self._key(f"nodes/{node_id}"))
243            return True
244        except Exception:
245            return False
246
247    def elect_leader(self, node_id: str, ttl_seconds: float = 30.0) -> bool:
248        if not self._client:
249            return False
250        try:
251            # Create a lease if not already held
252            if not self._lease_id:
253                lease = self._client.lease(int(ttl_seconds))
254                self._lease_id = lease.id
255
256            # Campaign: try to create the leader key with our lease (transaction)
257            leader_key = self._key("leader")
258            success, _ = self._client.transaction(
259                compare=[self._client.transactions.version(leader_key) == 0],
260                success=[self._client.transactions.put(leader_key, node_id, lease=self._lease_id)],
261                failure=[],
262            )
263            return success
264        except Exception:
265            return False
266
267    def get_leader(self) -> str | None:
268        if not self._client:
269            return None
270        try:
271            value, _ = self._client.get(self._key("leader"))
272            return value.decode() if value else None
273        except Exception:
274            return None
275
276    def watch_leader(self, callback: Callable[[str | None], None]) -> None:
277        if not self._client:
278            return
279        def _watch():
280            try:
281                events, cancel = self._client.watch(self._key("leader"))
282                for event in events:
283                    new_leader = event.value.decode() if event.value else None
284                    callback(new_leader)
285            except Exception:
286                pass
287        t = threading.Thread(target=_watch, daemon=True)
288        t.start()
289
290    def heartbeat(self, node_id: str) -> bool:
291        if not self._client or not self._lease_id:
292            return False
293        try:
294            self._client.refresh_lease(self._lease_id)
295            return True
296        except Exception:
297            return False
298
299    def list_nodes(self) -> list[CoordinationNode]:
300        if not self._client:
301            return []
302        nodes = []
303        try:
304            for value, _ in self._client.get_prefix(self._key("nodes/")):
305                data = json.loads(value.decode())
306                nodes.append(CoordinationNode(**{k: v for k, v in data.items() if k in CoordinationNode.__dataclass_fields__}))
307        except Exception:
308            pass
309        return nodes

320def create_coordination_backend(config: dict | None = None) -> CoordinationBackend:
321    """Create a coordination backend from config."""
322    cfg = config or {}
323    backend_type = cfg.get("type", "memory")
324    if backend_type not in BACKENDS:
325        raise ValueError(f"Unknown coordination backend: '{backend_type}'. Supported: {', '.join(BACKENDS.keys())}")
326    return BACKENDS[backend_type](cfg)
