 68class FileWatcher:
 69    """Watch a directory for file changes using polling.
 70
 71    Detects: created, modified, deleted files.
 72    Supports debouncing — multiple rapid changes → single event.
 73    """
 74
 75    def __init__(
 76        self, watch_dir: str, patterns: list[str] | None = None, poll_interval_ms: int = 500, debounce_ms: int = 300
 77    ):
 78        self.watch_dir = os.path.abspath(watch_dir)
 79        self.patterns = patterns or ["*.py"]
 80        self.poll_interval_ms = poll_interval_ms
 81        self.debounce_ms = debounce_ms
 82        self._running = False
 83        self._thread: threading.Thread | None = None
 84        self._file_states: dict[str, float] = {}  # filename → last modified
 85        self._changes: list[dict[str, Any]] = []
 86        self._lock = threading.RLock()
 87        self._callbacks: list[Callable] = []
 88
 89    def start(self) -> None:
 90        """Start watching the directory."""
 91        if self._running:
 92            return
 93        self._running = True
 94        # Initialize file states
 95        self._scan_directory()
 96        self._thread = threading.Thread(target=self._watch_loop, daemon=True)
 97        self._thread.start()
 98
 99    def stop(self) -> None:
100        """Stop watching."""
101        self._running = False
102        if self._thread and self._thread.is_alive():
103            self._thread.join(timeout=2.0)
104
105    def is_running(self) -> bool:
106        return self._running
107
108    def on_change(self, callback: Callable) -> None:
109        """Register a callback: callback(event_dict) when file changes."""
110        self._callbacks.append(callback)
111
112    def get_changes(self) -> list[dict[str, Any]]:
113        """Get and clear pending changes."""
114        with self._lock:
115            changes = list(self._changes)
116            self._changes.clear()
117        return changes
118
119    def _scan_directory(self) -> None:
120        """Scan directory and record file states."""
121        if not os.path.isdir(self.watch_dir):
122            return
123        try:
124            for entry in os.listdir(self.watch_dir):
125                filepath = os.path.join(self.watch_dir, entry)
126                if os.path.isfile(filepath) and self._matches_pattern(entry):
127                    self._file_states[entry] = os.path.getmtime(filepath)
128        except OSError:
129            pass
130
131    def _matches_pattern(self, filename: str) -> bool:
132        """Check if filename matches any watch pattern."""
133        import fnmatch
134
135        return any(fnmatch.fnmatch(filename, p) for p in self.patterns)
136
137    def _watch_loop(self) -> None:
138        """Polling loop."""
139        last_event_time: dict[str, float] = {}
140        while self._running:
141            try:
142                if not os.path.isdir(self.watch_dir):
143                    time.sleep(self.poll_interval_ms / 1000)
144                    continue
145
146                current_files: set[str] = set()
147                for entry in os.listdir(self.watch_dir):
148                    filepath = os.path.join(self.watch_dir, entry)
149                    if os.path.isfile(filepath) and self._matches_pattern(entry):
150                        current_files.add(entry)
151                        mtime = os.path.getmtime(filepath)
152
153                        if entry not in self._file_states:
154                            # New file
155                            self._record_change("created", entry, filepath)
156                            self._file_states[entry] = mtime
157                        elif mtime > self._file_states[entry]:
158                            # Modified
159                            now = time.time()
160                            last = last_event_time.get(entry, 0)
161                            if now - last > self.debounce_ms / 1000:
162                                self._record_change("modified", entry, filepath)
163                                last_event_time[entry] = now
164                            self._file_states[entry] = mtime
165
166                # Detect deleted files
167                for filename in list(self._file_states.keys()):
168                    if filename not in current_files:
169                        self._record_change("deleted", filename, os.path.join(self.watch_dir, filename))
170                        del self._file_states[filename]
171
172            except OSError:
173                pass
174
175            time.sleep(self.poll_interval_ms / 1000)
176
177    def _record_change(self, event_type: str, filename: str, filepath: str) -> None:
178        change = {
179            "type": event_type,
180            "filename": filename,
181            "filepath": filepath,
182            "timestamp": time.time(),
183        }
184        with self._lock:
185            self._changes.append(change)
186        for cb in self._callbacks:
187            try:
188                cb(change)
189            except Exception:
190                pass

196class HotReloadManager:
197    """Manages plugin version lifecycle and hot upgrades.
198
199    Each plugin can have multiple registered versions. The manager:
200      - Tracks which version is active
201      - Supports drain → upgrade → activate flow
202      - Maintains version history for rollback
203      - Supports canary deployment with traffic splitting
204    """
205
206    def __init__(self, upgrade_strategy: UpgradeStrategy = UpgradeStrategy.ROLLING):
207        self._versions: dict[str, dict[str, PluginVersion]] = {}  # plugin_id → {version → PluginVersion}
208        self._active: dict[str, str] = {}  # plugin_id → active_version
209        self.upgrade_strategy = upgrade_strategy
210        self._upgrade_history: list[dict[str, Any]] = []
211        self._lock = threading.RLock()
212
213    def register_version(
214        self, plugin_id: str, version: str, entrypoint: str, checksum: str = "", canary_percent: int = 0, **metadata
215    ) -> PluginVersion:
216        """Register a new plugin version. The latest version becomes active."""
217        pv = PluginVersion(
218            plugin_id=plugin_id,
219            version=version,
220            entrypoint=entrypoint,
221            checksum=checksum,
222            canary_percent=canary_percent,
223            metadata=metadata,
224            activated_at=time.time(),
225        )
226
227        with self._lock:
228            if plugin_id not in self._versions:
229                self._versions[plugin_id] = {}
230            self._versions[plugin_id][version] = pv
231            self._active[plugin_id] = version
232
233            self._upgrade_history.append(
234                {
235                    "plugin_id": plugin_id,
236                    "version": version,
237                    "action": "registered",
238                    "timestamp": time.time(),
239                }
240            )
241
242        return pv
243
244    def get_version(self, plugin_id: str, version: str) -> PluginVersion | None:
245        """Get a specific plugin version."""
246        versions = self._versions.get(plugin_id, {})
247        return versions.get(version)
248
249    def get_active_version(self, plugin_id: str) -> PluginVersion | None:
250        """Get the currently active version of a plugin."""
251        active_ver = self._active.get(plugin_id)
252        if active_ver:
253            return self._versions.get(plugin_id, {}).get(active_ver)
254        return None
255
256    def get_versions(self, plugin_id: str) -> list[PluginVersion]:
257        """List all versions of a plugin (oldest first)."""
258        versions = self._versions.get(plugin_id, {})
259        return sorted(versions.values(), key=lambda v: v.created_at)
260
261    def get_version_history(self, plugin_id: str) -> list[PluginVersion]:
262        """Get full version history (all versions, oldest first)."""
263        return self.get_versions(plugin_id)
264
265    def drain_version(self, plugin_id: str, version: str) -> bool:
266        """Mark a version as draining — stop routing new tasks to it."""
267        pv = self.get_version(plugin_id, version)
268        if not pv:
269            return False
270        pv.status = "drained"
271        pv.drained_at = time.time()
272
273        self._upgrade_history.append(
274            {
275                "plugin_id": plugin_id,
276                "version": version,
277                "action": "drained",
278                "timestamp": time.time(),
279            }
280        )
281        return True
282
283    def activate_version(self, plugin_id: str, version: str) -> bool:
284        """Manually activate a specific version."""
285        pv = self.get_version(plugin_id, version)
286        if not pv:
287            return False
288        pv.status = "active"
289        pv.activated_at = time.time()
290
291        with self._lock:
292            self._active[plugin_id] = version
293
294        self._upgrade_history.append(
295            {
296                "plugin_id": plugin_id,
297                "version": version,
298                "action": "activated",
299                "timestamp": time.time(),
300            }
301        )
302        return True
303
304    def rollback(self, plugin_id: str, target_version: str) -> bool:
305        """Rollback to a previous version."""
306        pv = self.get_version(plugin_id, target_version)
307        if not pv:
308            return False
309
310        current = self.get_active_version(plugin_id)
311        if current:
312            current.status = "rolled_back"
313
314        return self.activate_version(plugin_id, target_version)
315
316    def set_upgrade_strategy(self, strategy: UpgradeStrategy) -> None:
317        self.upgrade_strategy = strategy
318
319    def get_upgrade_history(self) -> list[dict[str, Any]]:
320        return list(self._upgrade_history)
321
322    def list_plugins(self) -> list[dict[str, Any]]:
323        """List all plugins with their active versions."""
324        result = []
325        for plugin_id in self._versions:
326            active = self.get_active_version(plugin_id)
327            all_versions = self.get_versions(plugin_id)
328            result.append(
329                {
330                    "plugin_id": plugin_id,
331                    "active_version": active.version if active else None,
332                    "version_count": len(all_versions),
333                    "strategy": self.upgrade_strategy.value,
334                }
335            )
336        return result
