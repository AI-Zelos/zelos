 47class Namespace:
 48    """Isolated resource container for a tenant.
 49
 50    Every namespace tracks its own:
 51      - Goals (goal IDs only — actual goals stored in Runtime)
 52      - Tasks (task IDs)
 53      - Agents (agent IDs)
 54    """
 55
 56    def __init__(self, namespace_id: str, name: str = "", quotas: ResourceQuota | None = None):
 57        self.namespace_id = namespace_id
 58        self.name = name or namespace_id
 59        self.quotas = quotas or ResourceQuota()
 60        self._goals: list[str] = []
 61        self._tasks: list[str] = []
 62        self._agents: list[str] = []
 63        self._created_at = time.time()
 64        self._lock = threading.RLock()
 65
 66    # ── Goal tracking ──
 67
 68    def add_goal(self, goal_id: str) -> bool:
 69        with self._lock:
 70            if len(self._goals) >= self.quotas.max_goals:
 71                return False
 72            self._goals.append(goal_id)
 73            return True
 74
 75    def remove_goal(self, goal_id: str) -> None:
 76        with self._lock:
 77            if goal_id in self._goals:
 78                self._goals.remove(goal_id)
 79
 80    # ── Task tracking ──
 81
 82    def add_task(self, task_id: str) -> bool:
 83        with self._lock:
 84            if len(self._tasks) >= self.quotas.max_tasks:
 85                return False
 86            self._tasks.append(task_id)
 87            return True
 88
 89    def remove_task(self, task_id: str) -> None:
 90        with self._lock:
 91            if task_id in self._tasks:
 92                self._tasks.remove(task_id)
 93
 94    # ── Agent tracking ──
 95
 96    def add_agent(self, agent_id: str) -> bool:
 97        with self._lock:
 98            if len(self._agents) >= self.quotas.max_agents:
 99                return False
100            self._agents.append(agent_id)
101            return True
102
103    def remove_agent(self, agent_id: str) -> None:
104        with self._lock:
105            if agent_id in self._agents:
106                self._agents.remove(agent_id)
107
108    # ── Quota Checks ──
109
110    def check_quota(self, resource_type: str) -> bool:
111        """Check if adding one more of resource_type is allowed."""
112        with self._lock:
113            if resource_type == "goals":
114                return len(self._goals) < self.quotas.max_goals
115            elif resource_type == "tasks":
116                return len(self._tasks) < self.quotas.max_tasks
117            elif resource_type == "agents":
118                return len(self._agents) < self.quotas.max_agents
119            elif resource_type == "concurrent_tasks":
120                return self.active_task_count() < self.quotas.max_concurrent_tasks
121            return True
122
123    def active_task_count(self) -> int:
124        return len(self._tasks)
125
126    # ── Counts ──
127

129    def goal_count(self) -> int:
130        with self._lock:
131            return len(self._goals)
132
133    @property
134    def task_count(self) -> int:

136            return len(self._tasks)

138    @property
139    def agent_count(self) -> int:

141            return len(self._agents)

144        return {

183class TenantManager:
184    """Central tenant registry with cross-tenant isolation enforcement.
185
186    Every tenant gets its own Namespace. Resources in one namespace
187    are invisible to other tenants.
188    """
189
190    def __init__(self):
191        self._tenants: dict[str, Tenant] = {}
192        self._default_namespace = Namespace("default", "Default Namespace")
193        self._lock = threading.RLock()
194
195    def register_tenant(
196        self, tenant_id: str, name: str = "", quotas: ResourceQuota | None = None, metadata: dict | None = None
197    ) -> Tenant:
198        """Register a new tenant with its own namespace."""
199        namespace = Namespace(tenant_id, name, quotas)
200        tenant = Tenant(
201            tenant_id=tenant_id,
202            name=name or tenant_id,
203            namespace=namespace,
204            metadata=metadata or {},
205        )
206        with self._lock:
207            self._tenants[tenant_id] = tenant
208        return tenant
209
210    def get_tenant(self, tenant_id: str) -> Tenant | None:
211        return self._tenants.get(tenant_id)
212
213    def get_namespace(self, tenant_id: str) -> Namespace | None:
214        """Get a tenant's namespace. Returns default if not found."""
215        tenant = self._tenants.get(tenant_id)
216        return tenant.namespace if tenant else self._default_namespace
217
218    def deactivate_tenant(self, tenant_id: str) -> bool:
219        """Deactivate a tenant — all operations rejected."""
220        tenant = self._tenants.get(tenant_id)
221        if not tenant:
222            return False
223        tenant.active = False
224        return True
225
226    def activate_tenant(self, tenant_id: str) -> bool:
227        """Reactivate a tenant."""
228        tenant = self._tenants.get(tenant_id)
229        if not tenant:
230            return False
231        tenant.active = True
232        return True
233
234    def is_active(self, tenant_id: str) -> bool:
235        tenant = self._tenants.get(tenant_id)
236        return tenant.active if tenant else True  # Default tenant always active
237
238    def remove_tenant(self, tenant_id: str) -> bool:
239        """Permanently remove a tenant and its namespace."""
240        with self._lock:
241            if tenant_id in self._tenants:
242                del self._tenants[tenant_id]
243                return True
244        return False
245
246    def list_tenants(self) -> list[Tenant]:
247        return list(self._tenants.values())
248
249    def tenant_count(self) -> int:
250        return len(self._tenants)
251
252    def get_usage_report(self) -> dict[str, Any]:
253        """Aggregate usage across all tenants."""
254        tenants_usage = {}
255        for tenant_id, tenant in self._tenants.items():
256            ns = tenant.namespace
257            tenants_usage[tenant_id] = {
258                "name": tenant.name,
259                "active": tenant.active,
260                "goals": ns.goal_count,
261                "tasks": ns.task_count,
262                "agents": ns.agent_count,
263                "quotas": ns.quotas.to_dict(),
264            }
265        return {
266            "total_tenants": len(self._tenants),
267            "active_tenants": sum(1 for t in self._tenants.values() if t.active),
268            "tenants": tenants_usage,
269        }
