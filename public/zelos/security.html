 40class AccessControl:
 41    """Role-Based Access Control with wildcard matching.
 42
 43    Default roles:
 44      - admin: * (all permissions — wildcard)
 45      - operator: goal.submit, goal.cancel, task.*, agent.read, plugin.*
 46      - agent: task.execute, agent.heartbeat, artifact.create
 47      - viewer: goal.read, task.read, agent.read, metrics.read
 48    """
 49
 50    DEFAULT_ROLES = {
 51        "admin": {
 52            "permissions": ["*"],
 53            "description": "Full system access — all actions allowed",
 54        },
 55        "operator": {
 56            "permissions": [
 57                "goal.submit",
 58                "goal.cancel",
 59                "goal.read",
 60                "task.create",
 61                "task.cancel",
 62                "task.read",
 63                "agent.read",
 64                "agent.register",
 65                "plugin.read",
 66                "plugin.configure",
 67                "metrics.read",
 68            ],
 69            "description": "Operational control — manage goals and tasks",
 70        },
 71        "agent": {
 72            "permissions": [
 73                "task.execute",
 74                "agent.heartbeat",
 75                "artifact.create",
 76            ],
 77            "description": "Agent execution — receive and complete tasks",
 78        },
 79        "viewer": {
 80            "permissions": [
 81                "goal.read",
 82                "task.read",
 83                "agent.read",
 84                "metrics.read",
 85            ],
 86            "description": "Read-only access to all resources",
 87        },

 90    def __init__(self):
 91        self.roles: dict[str, Role] = {}
 92        self._lock = threading.RLock()
 93        self._init_defaults()
 94
 95    def _init_defaults(self) -> None:
 96        for name, spec in self.DEFAULT_ROLES.items():
 97            self.add_role(name, spec["permissions"], spec["description"])
 98
 99    def add_role(self, name: str, permissions: list[str], description: str = "") -> Role:
100        """Add or replace a role."""
101        role = Role(name=name, permissions=set(permissions), description=description)
102        with self._lock:
103            self.roles[name] = role
104        return role
105
106    def update_role(
107        self, name: str, add_permissions: list[str] | None = None, remove_permissions: list[str] | None = None
108    ) -> Role | None:
109        """Update an existing role's permissions."""
110        with self._lock:
111            role = self.roles.get(name)
112            if not role:
113                return None
114            if add_permissions:
115                role.permissions.update(add_permissions)
116            if remove_permissions:
117                role.permissions.difference_update(remove_permissions)
118            return role
119
120    def remove_role(self, name: str) -> bool:
121        """Remove a role. Returns True if it existed."""
122        with self._lock:
123            if name in self.roles:
124                del self.roles[name]
125                return True
126            return False
127
128    def check(self, role_name: str, action: str) -> bool:
129        """Check if a role has permission for an action.
130
131        Supports:
132          - Exact match: "goal.submit" matches "goal.submit"
133          - Wildcard: role with "task.*" matches "task.create", "task.execute", etc.
134          - Super wildcard: role with "*" matches everything.
135        """
136        with self._lock:
137            role = self.roles.get(role_name)
138            if not role:
139                return False
140
141        # Direct wildcard → all access
142        if "*" in role.permissions:
143            return True
144
145        # Exact match
146        if action in role.permissions:
147            return True
148
149        # Pattern match: "task.*" matches "task.create"
150        action_parts = action.split(".")
151        for perm in role.permissions:
152            if perm.endswith(".*"):
153                prefix = perm[:-2]
154                perm_parts = prefix.split(".")
155                if action_parts[: len(perm_parts)] == perm_parts:
156                    return True
157
158        return False
159
160    def list_permissions(self, role_name: str) -> list[str]:
161        """List all permissions for a role."""
162        role = self.roles.get(role_name)
163        return sorted(role.permissions) if role else []
164
165    def list_roles(self) -> list[dict[str, Any]]:
166        """List all roles with their permissions."""
167        return [
168            {"name": r.name, "permissions": sorted(r.permissions), "description": r.description}
169            for r in self.roles.values()
170        ]

202class AuditLogger:
203    """Immutable, append-only audit log with multi-field query support.
204
205    Every security-sensitive operation MUST be logged:
206      - Goal submission / cancellation
207      - Task dispatch / completion / failure
208      - Agent registration / removal
209      - Permission changes
210      - API key usage
211    """
212
213    def __init__(self, max_events: int = 100000):
214        self._events: list[AuditEvent] = []
215        self._max_events = max_events
216        self._lock = threading.RLock()
217
218    def log(
219        self, actor: str, action: str, resource: str, detail: str = "", result: str = "success", **metadata
220    ) -> AuditEvent:
221        """Record an audit event. Thread-safe."""
222        event = AuditEvent(
223            event_id=str(uuid.uuid4()),
224            timestamp=time.time(),
225            actor=actor,
226            action=action,
227            resource=resource,
228            detail=detail,
229            result=result,
230            metadata=metadata,
231        )
232        with self._lock:
233            if len(self._events) >= self._max_events:
234                self._events.pop(0)  # Ring-buffer: drop oldest
235            self._events.append(event)
236        return event
237
238    def query(
239        self,
240        actor: str | None = None,
241        action: str | None = None,
242        resource: str | None = None,
243        result: str | None = None,
244        start_time: float | None = None,
245        end_time: float | None = None,
246        limit: int = 1000,
247    ) -> list[AuditEvent]:
248        """Query audit events with multiple filter dimensions."""
249        results = []
250        with self._lock:
251            for e in self._events:
252                if actor is not None and e.actor != actor:
253                    continue
254                if action is not None and e.action != action:
255                    continue
256                if resource is not None and e.resource != resource:
257                    continue
258                if result is not None and e.result != result:
259                    continue
260                if start_time is not None and e.timestamp < start_time:
261                    continue
262                if end_time is not None and e.timestamp > end_time:
263                    continue
264                results.append(e)
265                if len(results) >= limit:
266                    break
267        return results
268
269    def export_json(self) -> str:
270        """Export all audit events as JSON string."""
271        with self._lock:
272            return json.dumps([e.to_dict() for e in self._events], indent=2)
273
274    def export_json_file(self, filepath: str) -> int:
275        """Export to JSON file. Returns count."""
276        data = json.dumps([e.to_dict() for e in self._events], indent=2)
277        with open(filepath, "w") as f:
278            f.write(data)
279        return len(self._events)
280
281    def total_events(self) -> int:
282        return len(self._events)
283
284    def clear(self) -> None:
285        with self._lock:
286            self._events.clear()

305class APIKeyManager:
306    """API Key lifecycle management — generate, validate, revoke, expire.
307
308    Key format: zelos_<random_64_hex_chars>
309    Stored as SHA-256 hash — plaintext key is never persisted.
310    """
311
312    KEY_PREFIX = "zelos_"
313    KEY_BYTES = 32  # 256 bits of entropy
314
315    def __init__(
316        self,
317        max_failures: int = 10,
318        failure_window_seconds: float = 60.0,
319        auto_revoke: bool = True,
320    ):
321        self._keys: dict[str, _APIKeyEntry] = {}
322        self._lock = threading.RLock()
323        self._failed_attempts: dict[str, list[float]] = {}
324        self.max_failures = max_failures
325        self.failure_window_seconds = failure_window_seconds
326        self.auto_revoke = auto_revoke
327
328    def generate_key(self, role: str, description: str = "", ttl_seconds: float | None = None) -> str:
329        """Generate a new API key. Returns the plaintext key (show once!)."""
330        random_bytes = secrets.token_bytes(self.KEY_BYTES)
331        key_plaintext = self.KEY_PREFIX + random_bytes.hex()
332        key_hash = self._hash(key_plaintext)
333
334        expires_at = (time.time() + ttl_seconds) if ttl_seconds and ttl_seconds > 0 else None
335
336        entry = _APIKeyEntry(
337            key_hash=key_hash,
338            role=role,
339            description=description,
340            created_at=time.time(),
341            expires_at=expires_at,
342        )
343
344        with self._lock:
345            self._keys[key_hash] = entry
346
347        return key_plaintext
348
349    def validate(self, key_plaintext: str) -> dict[str, Any] | None:
350        """Validate an API key. Returns {role, description, ...} or None."""
351        if not key_plaintext or not key_plaintext.startswith(self.KEY_PREFIX):
352            return None
353
354        key_hash = self._hash(key_plaintext)
355
356        with self._lock:
357            entry = self._keys.get(key_hash)
358            if not entry:
359                return None
360            if entry.revoked:
361                return None
362            if entry.expires_at is not None and time.time() > entry.expires_at:
363                return None
364
365            entry.last_used_at = time.time()
366
367        return {
368            "role": entry.role,
369            "description": entry.description,
370            "created_at": entry.created_at,
371            "expires_at": entry.expires_at,
372            "last_used_at": entry.last_used_at,
373        }
374
375    def revoke(self, key_plaintext: str) -> bool:
376        """Revoke an API key. Returns True if found."""
377        key_hash = self._hash(key_plaintext)
378        with self._lock:
379            if key_hash in self._keys:
380                self._keys[key_hash].revoked = True
381                return True
382        return False
383
384    def list_keys(self) -> list[dict[str, Any]]:
385        """List all keys (hashes only, never plaintext)."""
386        with self._lock:
387            return [
388                {
389                    "key_hash": kh[:16] + "...",
390                    "role": e.role,
391                    "description": e.description,
392                    "created_at": e.created_at,
393                    "expires_at": e.expires_at,
394                    "revoked": e.revoked,
395                    "last_used_at": e.last_used_at,
396                }
397                for kh, e in self._keys.items()
398            ]
399
400    # ── Anomaly Detection (v0.5.0) ──
401
402    def _record_failure(self, key_plaintext: str) -> bool:
403        """Record a failed auth attempt. Returns True if key was auto-revoked."""
404        key_hash = self._hash(key_plaintext)
405        now = time.time()
406        with self._lock:
407            if key_hash not in self._failed_attempts:
408                self._failed_attempts[key_hash] = []
409            self._failed_attempts[key_hash].append(now)
410            # Prune old failures
411            cutoff = now - self.failure_window_seconds
412            self._failed_attempts[key_hash] = [t for t in self._failed_attempts[key_hash] if t > cutoff]
413            recent = len(self._failed_attempts[key_hash])
414            if self.auto_revoke and recent >= self.max_failures and key_hash in self._keys:
415                self._keys[key_hash].revoked = True
416                return True
417        return False
418
419    def get_failure_count(self, key_plaintext: str) -> int:
420        """Get recent failure count for a key."""
421        key_hash = self._hash(key_plaintext)
422        with self._lock:
423            return len(self._failed_attempts.get(key_hash, []))
424
425    @staticmethod
426    def _hash(key: str) -> str:
427        return hashlib.sha256(key.encode()).hexdigest()
