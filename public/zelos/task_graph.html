 58class TaskGraphEngine:
 59    """Kernel component — manages Tasks and their dependency DAG."""
 60
 61    def __init__(self):
 62        self._tasks: dict[str, Task] = {}
 63        self._dependencies: dict[str, set[str]] = {}  # task_id → {dependency_ids}
 64        self._dependents_map: dict[str, set[str]] = {}  # task_id → {dependent_ids}
 65        self._created_task_ids: set[str] = set()  # v0.7.0: O(1) lookup for created tasks
 66
 67    # ── Task CRUD ──
 68
 69    def add_task(self, task: Task) -> None:
 70        task.created_at = __import__("time").time()
 71        task.updated_at = task.created_at
 72        self._tasks[task.task_id] = task
 73        self._dependencies[task.task_id] = set(task.dependencies)
 74        self._created_task_ids.add(task.task_id)
 75        # Register as dependent on each dependency
 76        for dep_id in task.dependencies:
 77            self._dependents_map.setdefault(dep_id, set()).add(task.task_id)
 78
 79    def get_task(self, task_id: str) -> Task | None:
 80        return self._tasks.get(task_id)
 81
 82    def list_tasks(self) -> list[Task]:
 83        return list(self._tasks.values())
 84
 85    # ── State Transitions ──
 86
 87    def transition(self, task_id: str, to_status: TaskStatus, agent_id: str | None = None) -> Task:
 88        task = self._get_required(task_id)
 89        from_status = task.status
 90        valid = VALID_TRANSITIONS.get(from_status, set())
 91        if to_status not in valid:
 92            raise ValueError(f"Invalid transition: {from_status.value} → {to_status.value}")
 93
 94        task.status = to_status
 95        task.updated_at = __import__("time").time()
 96        if agent_id and to_status in (TaskStatus.ASSIGNED, TaskStatus.STARTED):
 97            task.assigned_agent_id = agent_id
 98        # v0.7.0: track created set for O(1) evaluate_all
 99        if from_status == TaskStatus.CREATED and to_status != TaskStatus.CREATED:
100            self._created_task_ids.discard(task_id)
101        elif to_status == TaskStatus.CREATED:
102            self._created_task_ids.add(task_id)
103        return task
104
105    def _get_required(self, task_id: str) -> Task:
106        if task_id not in self._tasks:
107            raise KeyError(f"Task not found: {task_id}")
108        return self._tasks[task_id]
109
110    # ── Dependency Resolution ──
111
112    def evaluate_dependencies(self, task_id: str) -> bool:
113        """Returns True if task becomes READY."""
114        task = self._get_required(task_id)
115        if task.status != TaskStatus.CREATED:
116            return False
117
118        for dep_id in task.dependencies:
119            dep = self._tasks.get(dep_id)
120            if dep is None or dep.status != TaskStatus.COMPLETED:
121                return False
122
123        self.transition(task_id, TaskStatus.READY)
124        return True
125
126    def evaluate_all(self) -> list[str]:
127        """Evaluate all CREATED tasks (O(|CREATED|) via _created_task_ids set). Returns task_ids that became READY."""
128        ready = []
129        for task_id in list(self._created_task_ids):
130            t = self._tasks.get(task_id)
131            if t and t.status == TaskStatus.CREATED and self.evaluate_dependencies(task_id):
132                ready.append(task_id)
133        return ready
134
135    def on_task_completed(self, completed_task_id: str) -> list[str]:
136        """Notify that a task completed. Evaluate its dependents. Returns newly READY task_ids."""
137        ready = []
138        for dep_id in self._dependents_map.get(completed_task_id, set()):
139            if dep_id in self._tasks and self._tasks[dep_id].status == TaskStatus.CREATED:
140                if self.evaluate_dependencies(dep_id):
141                    ready.append(dep_id)
142        return ready
143
144    def get_ready_tasks(self) -> list[Task]:
145        return [t for t in self._tasks.values() if t.status == TaskStatus.READY]
146
147    # ── DAG Validation ──
148
149    def add_dependency(self, from_task_id: str, to_task_id: str) -> None:
150        """Add edge: from → to (to depends on from)."""
151        # Check acyclicity
152        if self._would_create_cycle(from_task_id, to_task_id):
153            raise ValueError(f"Adding edge {from_task_id}→{to_task_id} would create a cycle")
154        to_task = self._get_required(to_task_id)
155        if from_task_id not in to_task.dependencies:
156            to_task.dependencies.append(from_task_id)
157            self._dependencies.setdefault(to_task_id, set()).add(from_task_id)
158            self._dependents_map.setdefault(from_task_id, set()).add(to_task_id)
159
160    def _would_create_cycle(self, from_id: str, to_id: str) -> bool:
161        """Check if adding edge from→to creates a cycle (DFS from to_id back to from_id)."""
162        if from_id == to_id:
163            return True
164        visited = set()
165
166        def dfs(current):
167            if current == from_id:
168                return True
169            if current in visited:
170                return False
171            visited.add(current)
172            for dep in self._dependents_map.get(current, set()):
173                if dfs(dep):
174                    return True
175            return False
176
177        return dfs(to_id)
178
179    # ── Dynamic Modification ──
180
181    def add_task_dynamic(self, task: Task) -> None:
182        """Add a new task to an existing plan."""
183        self.add_task(task)
184        # Evaluate immediately
185        self.evaluate_dependencies(task.task_id)
186
187    def remove_task(self, task_id: str) -> None:
188        """Remove an unstarted task."""
189        task = self._get_required(task_id)
190        if task.status in (TaskStatus.STARTED, TaskStatus.COMPLETED):
191            raise ValueError(f"Cannot remove task in '{task.status.value}' state")
192        # Remove dependency edges
193        for dep_id in task.dependencies:
194            self._dependents_map.get(dep_id, set()).discard(task_id)
195        for dep_id in self._dependents_map.get(task_id, set()):
196            dep_task = self._tasks.get(dep_id)
197            if dep_task:
198                dep_task.dependencies = [d for d in dep_task.dependencies if d != task_id]
199        self._tasks.pop(task_id, None)
200        self._dependencies.pop(task_id, None)
201        self._dependents_map.pop(task_id, None)
